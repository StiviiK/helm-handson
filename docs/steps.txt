# Update application
- Change something
- Build and redeploy it
- ./build-and-push.sh latest

# Create Helm Chart
helm create Helm2

# Patch Helm Chart
- Update Helm2/values.yaml from Helm/values.yaml
- Update ingress section
    - Update host and tls host
    - Update secretname
- Update deployment.yaml
    - Update http port to 8080

# Show Resources
helm upgrade --namespace default --install -f Helm2/values.yaml thromis-demo1 Helm2 --dry-run

# Deploy Helm Chart
helm upgrade --namespace default --install -f Helm2/values.yaml thromis-demo2 Helm2

# List Resources
kubectl get all -n default
